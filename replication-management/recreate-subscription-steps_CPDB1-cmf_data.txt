for database replication definition CPDB2_cmf_data_dbrep
with primary at CPDB2.cmf_data
with replicate at CPDB1.cmf_data
without purge;

resume connection to CPDB1.cmf_data;

define subscription CPDB2_cmf_data_dbsub
for database replication definition CPDB2_cmf_data_dbrep
with primary at CPDB2.cmf_data
with replicate at CPDB1.cmf_data
subscribe to truncate table
use dump marker;

validate subscription CPDB2_cmf_data_dbsub
for database replication definition CPDB2_cmf_data_dbrep
with primary at CPDB2.cmf_data
with replicate at CPDB1.cmf_data;

resume connection to CPDB1.cmf_data;
