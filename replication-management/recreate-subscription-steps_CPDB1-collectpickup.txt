for database replication definition CPDB2_collectpickup_dbrep
with primary at CPDB2.collectpickup
with replicate at CPDB1.collectpickup

for database replication definition CPDB2_collectpickup_dbrep
with primary at CPDB2.collectpickup
with replicate at CPDB1.collectpickup;
