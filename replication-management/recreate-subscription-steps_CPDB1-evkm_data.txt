define subscription CPDB2_evkm_data_dbsub
for database replication definition CPDB2_evkm_data_dbrep
with primary at CPDB2.evkm_data
with replicate at CPDB1.evkm_data
subscribe to truncate table
use dump marker;

for database replication definition CPDB2_evkm_data_dbrep
with primary at CPDB2.evkm_data
with replicate at CPDB1.evkm_data;
